import { RateLimiter } from '../utils/rate-limiter';
import { Logger } from '../utils/logger';
import { UserRepository } from '../repositories/user.repository';
import { TokenBlacklist } from '../utils/token-blacklist';
import { SecurityAuditLog } from '../utils/security-audit';
const logger = new Logger('AuthMiddleware');
const rateLimiter = new RateLimiter({ windowMs: 15 * 60 * 1000, max: 100 });
const tokenBlacklist = new TokenBlacklist();
const auditLog = new SecurityAuditLog();

export async function authMiddleware(
  req: Request,
  res: Response,
  next: NextFunction
): Promise<void> {
  const startTime = Date.now();
  const requestId = req.headers['x-request-id'] as string || `req-${Date.now()}`;

    const rateLimitResult = await rateLimiter.check(req.ip);
    if (!rateLimitResult.allowed) {
      logger.warn(`Rate limit exceeded for IP: ${req.ip}`, { requestId });
      res.setHeader('X-RateLimit-Limit', rateLimitResult.limit);
      res.setHeader('X-RateLimit-Remaining', 0);
      res.setHeader('X-RateLimit-Reset', rateLimitResult.resetTime);
      res.status(429).json({
        error: 'Too many requests',
        retryAfter: rateLimitResult.retryAfter
      });
      return;
    }

    const authHeader = req.headers['authorization'];
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      auditLog.record({ type: 'auth_failure', ip: req.ip, reason: 'missing_token', requestId });
      res.status(401).json({ error: 'Authorization header required' });
      return;
    }

    const token = authHeader.split(' ')[1];

    if (await tokenBlacklist.isBlacklisted(token)) {
      auditLog.record({ type: 'auth_failure', ip: req.ip, reason: 'blacklisted_token', requestId });
      res.status(401).json({ error: 'Token has been revoked' });
      return;
    }

    const decoded = jwt.verify(token, process.env.JWT_SECRET!) as JWTPayload;

    const user = await UserRepository.findById(decoded.userId);
    if (!user || !user.isActive) {
      auditLog.record({ type: 'auth_failure', ip: req.ip, reason: 'inactive_user', requestId });
      res.status(401).json({ error: 'User account is inactive' });
      return;
    }

    req.user = { ...decoded, roles: user.roles, permissions: user.permissions };
    req.requestId = requestId;

    auditLog.record({
      type: 'auth_success',
      userId: user.id,
      ip: req.ip,
      requestId,
      latencyMs: Date.now() - startTime
    });

    next();
    if (error instanceof jwt.TokenExpiredError) {
      auditLog.record({ type: 'auth_failure', ip: req.ip, reason: 'token_expired', requestId });
      res.status(401).json({ error: 'Token has expired', code: 'TOKEN_EXPIRED' });
    } else if (error instanceof jwt.JsonWebTokenError) {
      auditLog.record({ type: 'auth_failure', ip: req.ip, reason: 'invalid_token', requestId });
      res.status(401).json({ error: 'Invalid token', code: 'INVALID_TOKEN' });
    } else {
      logger.error('Unexpected auth error', { error, requestId });
      res.status(500).json({ error: 'Internal server error' });
    }

export async function optionalAuth(
  req: Request,
  res: Response,
  next: NextFunction
): Promise<void> {
  try {
    await authMiddleware(req, res, next);
  } catch {
    next();
  }
}
router.put('/profile', authMiddleware, async (req: Request, res: Response) => {
  const { name, bio } = req.body;
  if (!name || typeof name !== 'string' || name.trim().length < 2) {
    return res.status(400).json({ error: 'Name must be at least 2 characters' });
  }
  if (bio && bio.length > 500) {
    return res.status(400).json({ error: 'Bio must be under 500 characters' });
  }
